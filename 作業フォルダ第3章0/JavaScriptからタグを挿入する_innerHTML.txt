

'use strict';

function total(price)
{
    const tax =0.1;
    return price + price * tax;
}

console.log('コーヒーメーカーの値段は'+ total(1000)+'円(税込)です。');

//-----   <span>タグを挿入してCSSを反映させる  ---//
document.getElementById('output').innerHTML='コーヒーメーカーの値段は<span>'+total(1500)+'</span>円（税込）です。';
